<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| MCP client → image_op handler | Untrusted `op`, `provider`, `params`, `outputPath`, `outputDir` flow into the handler |
| Handler → filesystem | New writes under `<outputDir>/.runs/<runId>/` (artifact + manifest) |
| Server startup → filesystem | Sweep walks and deletes `.runs/run-*` entries before transport connects |
| Module import → process | Bottom-of-file `main()` invocation must not run on `import` from tests (would block on stdio or call `process.exit`) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-09 | Tampering | Manifest stores raw `params.input` (file path) and `params.prompt` (user text) | accept | Personal-MCP scope, manifest is in user's own directory. Documented in CLAUDE.md decisions and 06-RESEARCH.md "Security note for manifest params". No transmission of manifest. |
| T-06-10 | Information Disclosure | Capability error.message may leak API keys into manifest + trace.error | mitigate | Inherit Phase 5's existing `error instanceof Error ? error.message : String(error)` pattern; do NOT add stack traces to the response. (No new exposure surface beyond Phase 5; existing risk.) |
| T-06-11 | DoS | image_op handler crash mid-write leaves `<runDir>/n1.png.tmp` orphans | mitigate | `writeFileAtomic` (from 06-01) cleans up `.tmp` on error before rethrow. Status='in_progress' manifest exists so orphans are still attributable. Retention sweep eventually reaps. |

   ```typescript
   export interface ImageOpArgs {
     op: string;
     provider: string;
     params: Record<string, unknown>;
     outputPath?: string;
     outputDir?: string;
   }

   export async function handleImageOp(args: ImageOpArgs): Promise<{
     content: Array<{ type: 'text'; text: string }>;
   }> {
     const { op, provider, params, outputPath, outputDir } = args;
     const startedAt = Date.now();
     const runId = createRunId();
     let runDir: string;
     try {
       runDir = await resolveRunDir(runId);
     } catch (err) {
       const message = err instanceof Error ? err.message : String(err);
       return {
         content: [{
           type: 'text' as const,
           text: JSON.stringify({
             success: false,
             error: `Failed to create run directory: ${message}`,
             runId,
             trace: { runId, nodes: [] },
           }),
         }],
       };
     }

     const baseInvocation: RunManifest['invocation'] = {
       tool: 'image_op',
       op,
       provider,
       params: params as Record<string, unknown>,
       outputPath,
       outputDir,
     };

     // Write in-progress manifest first (Pitfall 6: orphan PNGs without manifest are unfindable)
     await writeManifest(runDir, {
       schemaVersion: 1,
       runId,
       startedAt: new Date(startedAt).toISOString(),
       status: 'in_progress',
       invocation: baseInvocation,
       nodes: [],
     });

     const capability = capabilityRegistry.get(op as CapabilityOp, provider);
     if (!capability) {
       const endedAt = Date.now();
       const errMsg = `Capability not registered for op '${op}' and provider '${provider}'`;
       const errorNode = buildTraceNode({
         id: 'n1', op, provider,
         startedAtMs: startedAt, endedAtMs: endedAt,
         outcome: 'error', error: errMsg,
       });
       await writeManifest(runDir, {
         schemaVersion: 1,
         runId,
         startedAt: new Date(startedAt).toISOString(),
         endedAt: new Date(endedAt).toISOString(),
         status: 'error',
         invocation: baseInvocation,
         nodes: [{
           id: errorNode.id, op, provider,
           outcome: 'error', error: errMsg,
           durationMs: errorNode.durationMs,
         }],
         totalDurationMs: errorNode.durationMs,
         error: errMsg,
       });
       return {
         content: [{
           type: 'text' as const,
           text: JSON.stringify({
             success: false,
             error: errMsg,
             available: capabilityRegistry.list(op as CapabilityOp).map((c) => c.provider),
             runId,
             trace: { runId, nodes: [errorNode] },
           }),
         }],
       };
     }

     const nodeId = '1';
     const artifactPath = nodeArtifactPath(runDir, nodeId);

     try {
       validateCapabilityParams(capability, params);
       const nodeStartedAt = Date.now();
       const result = await capability.invoke({ params, outputPath, outputDir });
       const nodeEndedAt = Date.now();

       // 1. Persist run artifact (atomic)
       await writeFileAtomic(artifactPath, result.buffer);

       // 2. Save user-facing copy via existing rules (unchanged from Phase 5)
       const filePath = await resolveOutputPath({
         outputPath,
         outputDir,
         prompt: `${op}-${provider}`,
         provider,
       });
       await saveImage(result.buffer, filePath);

       const node = buildTraceNode({
         id: `n${nodeId}`,
         op, provider,
         model: result.model,
         artifactPath,
         output: filePath,
         startedAtMs: nodeStartedAt,
         endedAtMs: nodeEndedAt,
         outcome: 'success',
         revisedPrompt: result.revisedPrompt,
         metadata: result.metadata,
       });

       const endedAt = Date.now();
       await writeManifest(runDir, {
         schemaVersion: 1,
         runId,
         startedAt: new Date(startedAt).toISOString(),
         endedAt: new Date(endedAt).toISOString(),
         status: 'success',
         invocation: baseInvocation,
         nodes: [{
           id: node.id, op, provider,
           model: result.model,
           artifactPath, durationMs: node.durationMs, outcome: 'success',
         }],
         finalOutput: filePath,
         totalDurationMs: endedAt - startedAt,
       });

       return {
         content: [{
           type: 'text' as const,
           text: JSON.stringify({
             success: true,
             output: filePath,
             runId,
             trace: { runId, nodes: [node] },
           }),
         }],
       };
     } catch (error) {
       const endedAt = Date.now();
       const message = error instanceof Error ? error.message : String(error);
       const errorNode = buildTraceNode({
         id: `n${nodeId}`, op, provider,
         startedAtMs: startedAt, endedAtMs: endedAt,
         outcome: 'error', error: message,
       });
       await writeManifest(runDir, {
         schemaVersion: 1,
         runId,
         startedAt: new Date(startedAt).toISOString(),
         endedAt: new Date(endedAt).toISOString(),
         status: 'error',
         invocation: baseInvocation,
         nodes: [{
           id: errorNode.id, op, provider,
           outcome: 'error', error: message, durationMs: errorNode.durationMs,
         }],
         totalDurationMs: errorNode.durationMs,
         error: message,
       });
       return {
         content: [{
           type: 'text' as const,
           text: JSON.stringify({
             success: false,
             error: message,
             runId,
             trace: { runId, nodes: [errorNode] },
           }),
         }],
       };
     }
   }
   ```
